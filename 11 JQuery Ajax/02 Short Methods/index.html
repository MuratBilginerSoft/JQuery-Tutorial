AJAX Kısa Yöntemler

jQuery'nin pratik AJAX kısayollarını keşfedin

$.get() Metodu


// Basit GET isteği
$.get('https://jsonplaceholder.typicode.com/posts/1', 
    function(data) {
        console.log('Data:', data);
    }
);

// Parametreli GET isteği
$.get('https://jsonplaceholder.typicode.com/posts', {
    userId: 1,
    _limit: 2
}, function(data) {
    console.log('Posts:', data);
});

$.post() Metodu


// Basit POST isteği
$.post('https://jsonplaceholder.typicode.com/posts',
    {
        title: 'Yeni Gönderi',
        body: 'Gönderi içeriği',
        userId: 1
    },
    function(response) {
        console.log('Created:', response);
    }
);

$.getJSON() Metodu


// JSON verisi alma
$.getJSON('https://jsonplaceholder.typicode.com/users/1',
    function(data) {
        console.log('User data:', data);
    }
);

// Parametreli JSON isteği
$.getJSON('https://jsonplaceholder.typicode.com/comments', {
    postId: 1,
    _limit: 2
}, function(data) {
    console.log('Comments:', data);
});

$.getScript() Metodu


// Harici script yükleme
$.getScript('https://cdnjs.cloudflare.com/ajax/libs/moment.js/2.29.1/moment.min.js',
    function() {
        console.log('Moment.js yüklendi');
        console.log('Şu an:', moment().format('LLLL'));
    }
);
